CREATE TABLE Users(
    UID INT(10) PRIMARY KEY AUTO_INCREMENT NOT NULL,
    Username VARCHAR(50) NOT NULL,
    Password VARCHAR(50) NOT NULL,
    Forename VARCHAR(50) NOT NULL,
    Surname VARCHAR(50) NOT NULL,
    Email VARCHAR(50) NOT NULL,
    Firstad VARCHAR(50) NOT NULL,
    Secondad VARCHAR(50) NOT NULL
    );
CREATE TABLE Bookings(
    BookNo INT(11) PRIMARY KEY AUTO_INCREMENT NOT NULL,
    Username VARCHAR(50) NOT NULL,
    CollDate VARCHAR(13) NOT NULL,
    CollTime VARCHAR (10),
    DeliDate VARCHAR(13) NOT NULL,
    DeliTime VARCHAR (10)
    );
